	ORG	 0x0
V0:	WORD	$DEFAULT, 0x180
V1:	WORD	$INT1, 0x180
V2:	WORD	$INT2, 0x180
DEFAULT:	IRET
DOP:	WORD	0xFFF8


	ORG	0x00B
X:	WORD	0
ODZ_MIN:	WORD	0xFFE7
ODZ_MAX:	WORD	0x001A

	ORG	0x010

START:	DI
	CLA
	LD	#8
	OUT	7
	LD	#9
	OUT	3
	LD	#0xA
	OUT	5

	EI
IT:	CALL	ODZ_CHECK
	LD	X
	ADD	#2
	ST	X
	BR	IT

INT1:	PUSH	; Вывод функции 5X-2
	CALL	ODZ_CHECK
	LD	X
N1:	NOP	; Значение X
	ASL
	ASL
	ADD 	X;
	OUT	2;
N2:	NOP	; Значение 5X-2
	POP
	IRET

INT2:	PUSH	;мАСКИРОВАНИЕ, ТРА ТА ТА
	CALL	ODZ_CHECK;
	LD	X
N3:	NOP	;
	IN	4
	OR	DOP
	AND	X
	ST	X

N4:	NOP
	CALL	ODZ_CHECK
	POP
	IRET

ODZ_CHECK:	LD	X
	CMP	ODZ_MIN
	BLT	REPAINT
	CMP	ODZ_MAX
	BGE	REPAINT

REPAINT:	LD	ODZ_MIN
	ST	X
	RET
